FUNCTION ZJBOSS_PARAM_TEST.
*"----------------------------------------------------------------------
*"*"Local Interface:
*"  IMPORTING
*"     VALUE(IMPORT_CHAR_PARAM) TYPE  ZJBOSS_CHAR
*"     VALUE(IMPORT_NUM_PARAM) TYPE  ZJBOSS_NUM
*"     VALUE(IMPORT_INT_PARAM) TYPE  I
*"     VALUE(IMPORT_FLOAT_PARAM) TYPE  F
*"     VALUE(IMPORT_BCD_PARAM) TYPE  ZJBOSS_BCD
*"     VALUE(IMPORT_BINARY_PARAM) TYPE  ZJBOSS_BINARY
*"     VALUE(IMPORT_BINARY_ARRAY_PARAM) TYPE  ZJBOSS_BINARY_ARRAY
*"     VALUE(IMPORT_DATE_PARAM) TYPE  D
*"     VALUE(IMPORT_TIME_PARAM) TYPE  T
*"     VALUE(IMPORT_STRING_PARAM) TYPE  STRING
*"     VALUE(IMPORT_STRUCTURE_PARAM) TYPE  ZJBOSS_STRUCTURE
*"  EXPORTING
*"     VALUE(EXPORT_CHAR_PARAM) TYPE  ZJBOSS_CHAR
*"     VALUE(EXPORT_NUM_PARAM) TYPE  ZJBOSS_NUM
*"     VALUE(EXPORT_INT_PARAM) TYPE  I
*"     VALUE(EXPORT_FLOAT_PARAM) TYPE  F
*"     VALUE(EXPORT_BCD_PARAM) TYPE  ZJBOSS_BCD
*"     VALUE(EXPORT_BINARY_PARAM) TYPE  ZJBOSS_BINARY
*"     VALUE(EXPORT_BINARY_ARRAY_PARAM) TYPE  ZJBOSS_BINARY_ARRAY
*"     VALUE(EXPORT_DATE_PARAM) TYPE  D
*"     VALUE(EXPORT_TIME_PARAM) TYPE  T
*"     VALUE(EXPORT_STRING_PARAM) TYPE  STRING
*"     VALUE(EXPORT_STRUCTURE_PARAM) TYPE  ZJBOSS_STRUCTURE
*"  TABLES
*"      TABLE_TABLE_PARAM STRUCTURE  ZJBOSS_LINE_TYPE
*"  CHANGING
*"     VALUE(CHANGING_CHAR_PARAM) TYPE  ZJBOSS_CHAR
*"     VALUE(CHANGING_NUM_PARAM) TYPE  ZJBOSS_NUM
*"     VALUE(CHANGING_INT_PARAM) TYPE  I
*"     VALUE(CHANGING_FLOAT_PARAM) TYPE  F
*"     VALUE(CHANGING_BCD_PARAM) TYPE  ZJBOSS_BCD
*"     VALUE(CHANGING_BINARY_PARAM) TYPE  ZJBOSS_BINARY
*"     VALUE(CHANGING_BINARY_ARRAY_PARAM) TYPE  ZJBOSS_BINARY_ARRAY
*"     VALUE(CHANGING_DATE_PARAM) TYPE  D
*"     VALUE(CHANGING_TIME_PARAM) TYPE  T
*"     VALUE(CHANGING_STRING_PARAM) TYPE  STRING
*"     VALUE(CHANGING_STRUCTURE_PARAM) TYPE  ZJBOSS_STRUCTURE
*"  EXCEPTIONS
*"      CHANGING_CHAR_PARAM_EX
*"      CHANGING_NUM_PARAM_EX
*"      CHANGING_INT_PARAM_EX
*"      CHANGING_FLOAT_PARAM_EX
*"      CHANGING_BCD_PARAM_EX
*"      CHANGING_BINARY_PARAM_EX
*"      CHANGING_BINARY_ARRAY_PARAM_EX
*"      CHANGING_DATE_PARAM_EX
*"      CHANGING_TIME_PARAM_EX
*"      CHANGING_STRING_PARAM_EX
*"      CHANGING_STRUCTURE_CHAR_EX
*"      CHANGING_STRUCTURE_NUM_EX
*"      CHANGING_STRUCTURE_INT_EX
*"      CHANGING_STRUCTURE_FLOAT_EX
*"      CHANGING_STRUCTURE_BCD_EX
*"      CHANGING_STRUCTURE_BINARY_EX
*"      CHANGING_STRUCTURE_BIN_ARRY_EX
*"      CHANGING_STRUCTURE_DATE_EX
*"      CHANGING_STRUCTURE_TIME_EX
*"      CHANGING_STRUCTURE_STR_EX
*"----------------------------------------------------------------------

*-----------------------------------------------------------------------
* Data Type Definitions:
*
* ZJBOSS_CHAR         : C(10)
* ZJBOSS_NUM          : N(10)
* ZJBOSS_BCD          : P(9) DECIMALS 14
* ZJBOSS_INT          : I(4)
* ZJBOSS_DATE         : D(8)
* ZJBOSS_TIME         : T(6)
* ZJBOSS_FLOAT        : F(8)
* ZJBOSS_BINARY       : X(1)
* ZJBOSS_BINARY_ARRAY : X(10)
* ZJBOSS_STRING       : STRING
* ZJBOSS_STRUCTURE    :
*  CHAR_PARAM         : ZJBOSS_CHAR
*  NUM_PARAM          : ZJBOSS_NUM
*  INT_PARAM          : ZJBOSS_INT
*  FLOAT_PARAM        : ZJBOSS_FLOAT
*  BCD_PARAM          : ZJBOSS_BCD
*  BINARY_PARAM       : ZJBOSS_BINARY
*  BINARY_ARRAY_PARAM : ZJBOSS_BINARY_ARRAY
*  DATE_PARAM         : ZJBOSS_DATE
*  TIME_PARAM         : ZJBOSS_TIME
*  STRING_PARAM       : ZJBOSS_STRING
* ZJBOSS_LINE_TYPE    :
*  CHAR_PARAM         : ZJBOSS_CHAR
*  NUM_PARAM          : ZJBOSS_NUM
*  INT_PARAM          : ZJBOSS_INT
*  FLOAT_PARAM        : ZJBOSS_FLOAT
*  BCD_PARAM          : ZJBOSS_BCD
*  BINARY_PARAM       : ZJBOSS_BINARY
*  BINARY_ARRAY_PARAM : ZJBOSS_BINARY_ARRAY
*  DATE_PARAM         : ZJBOSS_DATE
*  TIME_PARAM         : ZJBOSS_TIME
*
*-----------------------------------------------------------------------

*
* Assign the values of import parameters to their corresponding export parameters
*

EXPORT_CHAR_PARAM = IMPORT_CHAR_PARAM .
EXPORT_NUM_PARAM = IMPORT_NUM_PARAM .
EXPORT_INT_PARAM = IMPORT_INT_PARAM .
EXPORT_FLOAT_PARAM = IMPORT_FLOAT_PARAM .
EXPORT_BCD_PARAM = IMPORT_BCD_PARAM .
EXPORT_BINARY_PARAM = IMPORT_BINARY_PARAM .
EXPORT_BINARY_ARRAY_PARAM = IMPORT_BINARY_ARRAY_PARAM .
EXPORT_DATE_PARAM = IMPORT_DATE_PARAM .
EXPORT_TIME_PARAM = IMPORT_TIME_PARAM .
EXPORT_STRING_PARAM = IMPORT_STRING_PARAM .
EXPORT_STRUCTURE_PARAM = IMPORT_STRUCTURE_PARAM .

*
* Check the imported values of changing parameters and export new values.
*

IF CHANGING_CHAR_PARAM = 'ABCDEFGHIJ' .
  CHANGING_CHAR_PARAM = 'ZYXWVUTSRQ' .
ELSE.
  RAISE CHANGING_CHAR_PARAM_EX .
ENDIF.

IF CHANGING_NUM_PARAM = '0123456789' .
  CHANGING_NUM_PARAM = '9876543210' .
ELSE.
  RAISE CHANGING_NUM_PARAM_EX .
ENDIF.

IF CHANGING_INT_PARAM = 1968526677 .
  CHANGING_INT_PARAM = 2058005162 .
ELSE.
  RAISE CHANGING_INT_PARAM_EX .
ENDIF.

IF CHANGING_FLOAT_PARAM = '1E38' .
  CHANGING_FLOAT_PARAM  = '2E38' .
ELSE.
  RAISE CHANGING_FLOAT_PARAM_EX .
ENDIF.

IF CHANGING_BCD_PARAM = '100.00000000000001' .
  CHANGING_BCD_PARAM  = '200.00000000000002' .
ELSE.
  RAISE CHANGING_BCD_PARAM_EX .
ENDIF.

IF CHANGING_BINARY_PARAM = 85 .
  CHANGING_BINARY_PARAM  = 170 .
ELSE.
  RAISE CHANGING_BINARY_PARAM_EX .
ENDIF.

IF CHANGING_BINARY_ARRAY_PARAM = 'FF0F1E2D3C4B5A607988' .
  CHANGING_BINARY_ARRAY_PARAM  = '8879605A4B3C2D1E0FFF' .
ELSE.
  RAISE CHANGING_BINARY_ARRAY_PARAM_EX .
ENDIF.

IF CHANGING_DATE_PARAM = '18610412' .
  CHANGING_DATE_PARAM  = '18650409' .
ELSE.
  RAISE CHANGING_DATE_PARAM_EX .
ENDIF.

IF CHANGING_TIME_PARAM = '121530' .
  CHANGING_TIME_PARAM  = '234515' .
ELSE.
  RAISE CHANGING_TIME_PARAM_EX .
ENDIF.

IF CHANGING_STRING_PARAM = 'Four score and seven years ago ...' .
  CHANGING_STRING_PARAM  = '... shall not perish from this earth.' .
ELSE.
  RAISE CHANGING_STRING_PARAM_EX .
ENDIF.

IF CHANGING_STRUCTURE_PARAM-CHAR_PARAM = 'ABCDEFGHIJ' .
  CHANGING_STRUCTURE_PARAM-CHAR_PARAM = 'ZYXWVUTSRQ' .
ELSE.
  RAISE CHANGING_STRUCTURE_CHAR_EX .
ENDIF.

IF CHANGING_STRUCTURE_PARAM-NUM_PARAM = '0123456789' .
  CHANGING_STRUCTURE_PARAM-NUM_PARAM = '9876543210' .
ELSE.
  RAISE CHANGING_STRUCTURE_NUM_EX .
ENDIF.

IF CHANGING_STRUCTURE_PARAM-INT_PARAM = 1968526677 .
  CHANGING_STRUCTURE_PARAM-INT_PARAM = 2058005162 .
ELSE.
  RAISE CHANGING_STRUCTURE_INT_EX .
ENDIF.

IF CHANGING_STRUCTURE_PARAM-FLOAT_PARAM = '1E38' .
  CHANGING_STRUCTURE_PARAM-FLOAT_PARAM  = '2E38' .
ELSE.
  RAISE CHANGING_STRUCTURE_FLOAT_EX .
ENDIF.

IF CHANGING_STRUCTURE_PARAM-BCD_PARAM = '100.00000000000001' .
  CHANGING_STRUCTURE_PARAM-BCD_PARAM  = '200.00000000000002' .
ELSE.
  RAISE CHANGING_STRUCTURE_BCD_EX .
ENDIF.

IF CHANGING_STRUCTURE_PARAM-BINARY_PARAM = 85 .
  CHANGING_STRUCTURE_PARAM-BINARY_PARAM  = 170 .
ELSE.
  RAISE CHANGING_STRUCTURE_BINARY_EX .
ENDIF.


IF CHANGING_STRUCTURE_PARAM-BINARY_ARRAY_PARAM = 'FF0F1E2D3C4B5A607988' .
  CHANGING_STRUCTURE_PARAM-BINARY_ARRAY_PARAM  = '8879605A4B3C2D1E0FFF' .
ELSE.
  RAISE CHANGING_STRUCTURE_BIN_ARRY_EX .
ENDIF.

IF CHANGING_STRUCTURE_PARAM-DATE_PARAM = '18610412' .
  CHANGING_STRUCTURE_PARAM-DATE_PARAM  = '18650409' .
ELSE.
  RAISE CHANGING_STRUCTURE_BIN_ARRY_EX .
ENDIF.

IF CHANGING_STRUCTURE_PARAM-TIME_PARAM = '121530' .
  CHANGING_STRUCTURE_PARAM-TIME_PARAM  = '234515' .
ELSE.
  RAISE CHANGING_STRUCTURE_TIME_EX .
ENDIF.

IF CHANGING_STRUCTURE_PARAM-STRING_PARAM = 'Four score and seven years ago ...' .
  CHANGING_STRUCTURE_PARAM-STRING_PARAM  = '... shall not perish from this earth.' .
ELSE.
  RAISE CHANGING_STRUCTURE_STR_EX .
ENDIF.

*
* Iterate through rows of test table and set new values.
*

LOOP AT TABLE_TABLE_PARAM.

  TABLE_TABLE_PARAM-CHAR_PARAM = 'ZYXWVUTSRQ' .
  TABLE_TABLE_PARAM-NUM_PARAM = '9876543210' .
  TABLE_TABLE_PARAM-INT_PARAM = 2058005162 .
  TABLE_TABLE_PARAM-FLOAT_PARAM  = '2E38' .
  TABLE_TABLE_PARAM-FLOAT_PARAM = '2E38' .
  TABLE_TABLE_PARAM-BCD_PARAM  = '200.00000000000002' .
  TABLE_TABLE_PARAM-BINARY_PARAM  = 170 .
  TABLE_TABLE_PARAM-BINARY_ARRAY_PARAM  = '8879605A4B3C2D1E0FFF' .
  TABLE_TABLE_PARAM-DATE_PARAM  = '18650409' .
  TABLE_TABLE_PARAM-TIME_PARAM  = '234515' .
  MODIFY TABLE_TABLE_PARAM.

ENDLOOP.


ENDFUNCTION.
